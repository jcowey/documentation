     # File app/controllers/identifiers_controller.rb, line 124
124:     def insert_error_here(content, line, column)
125:       # this routine is to place the error message below in the Leiden+ or XML returned when a parse error
126:       # occurs by taking the line and column from the message and giving the user the place in the content
127:       # the parse error occured in xsugars processing - may or may not be where the real error is depending
128:       # on what the error is - this processing is by character because there are multiple byte characters
129:       # possible in the text and a way to place msg with taking that into account
130:       #
131:       # line starts at 1 because first character is on first line before incrementing in loop
132:       # same logic for column, already on first character before incrementing in loop 
133:       # 'col' check has to come before 'new line' check in case error is on last char in the line
134:       line_cnt = 1
135:       col_cnt = 1
136:       content_error_here = ''
137:       content.each_char do |i|
138:         if line_cnt == line
139:           if col_cnt == column
140:             content_error_here << "**POSSIBLE ERROR**"
141:           end
142:           col_cnt += 1
143:         end
144:         if i == "\n"
145:           line_cnt += 1
146:         end
147:         content_error_here << i
148:       end
149:       return content_error_here
150:     end
